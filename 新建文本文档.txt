    loadFileProxy(file){
        return new Promise((resolve, reject)=>{
            const fr = new FileReader();
            fr.readAsDataURL(file);
            let fileContent = null;
            fr.onload = ()=> {
                fileContent = this.b64DecodeUnicode(fr.result);
               resolve(fileContent);
            };
        })
    }
    //校验文件是否重复且未修改
    checkeRepate = async (oldList,file)=> {
        let result = false;
        await this.loadFileProxy(file).then(result=>{
            oldList.map(async (old)=>{
                if(old){
                    if (file.name.substring(0, file.name.indexOf('.')) == old.taskName){
                        result = result==old.taskScript;
                    }
                }
            })
        })
        return result;
    }
